function openTab(evt, cityName) {
    var i, tabcontent, tablinks;
    tabcontent = document.getElementsByClassName("tabcontent");
    for (i = 0; i < tabcontent.length; i++) {
        tabcontent[i].style.display = "none";
    }
    tablinks = document.getElementsByClassName("tablinks");
    for (i = 0; i < tablinks.length; i++) {
        tablinks[i].className = tablinks[i].className.replace(" active", "");
    }
    document.getElementById(cityName).style.display = "block";
    evt.currentTarget.className += " active";
}

      function lordGradeStars(type){		
        switch(type){
         case '5':
         $('#star_cont').hide("slow");				
         $('#grade_cont').show();
         $('#grade_type').hide("slow");	
         break;

         case '4':
         $('#grade_cont').hide("slow");			
         $('#star_cont').show();
         $('#grade_type').hide("slow");	
         break;	

         case '3': case '6': case '7': case '9': case '10': case '11':     
         $('#grade_cont').hide("slow");			
         $('#star_cont').hide("slow");		
         $('#grade_type').show();
         break;

         default :
         $('#grade_cont').hide("slow");			
         $('#star_cont').hide("slow");		
         $('#grade_type').hide("slow");
         break;
       }
     }

function moduleFindPlace(){	
	
	var url = 'index.php?route=travel/tostay';
	
  var is_expired=$("#is_expired").is(":checked");
  
	var hotel_name = $('#hotel_name').val();	
  var res = hotel_name.replace(/(.*?) and.*/i, "$1") ;
  var res1 = res.replace(/(.*?) or.*/i, "$1") ;
  
  if (hotel_name != '' && hotel_name != 'Enter a Hotel or Place to search' && hotel_name != 'Search' ) {
    url += '&hotel_name=' + encodeURIComponent(res1);
  }

  var hotel_type = $('select[name=\'hotel_type\']').val();	
  if (hotel_type != '*') {
    url += '&hotel_type=' + encodeURIComponent(hotel_type);
  }

  var hotel_district = $('select[name=\'hotel_district\']').val();	
  if (hotel_district != '*') {
    url += '&hotel_district=' + encodeURIComponent(hotel_district);
  }

  var grade = 1;
  switch(hotel_type){
   case '5':
   grade = 1;	
   break;

   case '4':
   grade = 2;		
   break;	

   case '3': case '6': case '7': case '9': case '10': case '11':     
   grade = 3;	
   break;

   default :
   grade = 4;	
   break;
 }

 if(grade == 3){	
  var hotel_grade_type = $('select[name=\'hotel_grade_type\']').val();	
  if (hotel_grade_type != '*') {
   url += '&hotel_grade=' + encodeURIComponent(hotel_grade_type);
 }
} else if(grade == 1){	
  var hotel_grade = $('select[name=\'hotel_grade\']').val();	
  if (hotel_grade != '*') {
   url += '&hotel_grade=' + encodeURIComponent(hotel_grade);
 }
} else if(grade == 2) {
  var hotel_star = $('select[name=\'hotel_star\']').val();	
  if (hotel_star != '*') {
   url += '&hotel_star=' + encodeURIComponent(hotel_star);
 }	
}

url += '&is_expired=' + encodeURIComponent(is_expired);
